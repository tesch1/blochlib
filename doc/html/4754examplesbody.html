Using the ScriptParse class for simple print operations
This example is quite basic, and does not do much becuase this class is meant to be a superclass to other objects.
------the input file------
myscript{
   B=2
   a=B

   loop(i=5:10)
    print(a*i)
    a=a+1
    if(a==4)
      print(77)
    end
   end

}

-----end Input file----
#include "blochlib.h"

//the required 2 namespaces
using namespace BlochLib;
using namespace std;

int main(int argc, char **argc){
  
   std::string fname;
   query_parameter(argc,argv,1, "Input File Name: ", fname);
   Parameters pset(fname);

   ScriptParse myP;
   myP.parse(pset);
}

//-----This should print out-----
//a*i=10
//a*i=18
//a*i=28
//77=77
//a*i=40
//a*i=54
//a*i=70
